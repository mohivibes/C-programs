#include <stdio.h>

int main() {
int a[10][10],b[10][10],c1[10][10],r,c,i,j;
printf("How many rows and columns do you want to enter: ");
scanf("%d %d",&r,&c);
printf("Enter Matrix1 : ");
for(i=0;i<r;i++)
{
    for(j=0;j<c;j++)
{
    scanf("%d",&a[i][j]);
}
    
}
printf("Enter Matrix2 : ");
for(i=0;i<r;i++)
{
    for(j=0;j<c;j++)
{
    scanf("%d",&b[i][j]);
}
    
}
for(i=0;i<r;i++)
{
    for(j=0;j<c;j++)
{
    c1[i][j]=a[i][j]+b[i][j];
}
}
printf("Addition is :\n ");
for(i=0;i<r;i++)
{
    for(j=0;j<c;j++)
{
    printf("%d\t",c1[i][j]);
}
    printf("\n");
}
return 0;}
